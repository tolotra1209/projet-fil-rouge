# Déclaration des types d'enregistrements
Type1 
    parametre = Enregistrement
                    langue:chaine de caractere
                    couleur:couleur
                    preference:chaine de caractere
                Finenreg

Type2
    couleur = Enregistrement
                    choix1:chaine de caractere
                    choix2:chaine de caractere
                    choix3:chaine de caractere
              Finenreg

# Déclaration des variables
variable
    parametreUtilisateur: parametre
    langueParDefaut='Anglais'
    choix1ParDefaut='Rouge'
    choix2ParDefaut='Bleue'
    choix3ParDefaut='Verte'
    couleurParDefaut=[choix1ParDefaut, choix2ParDefaut, choix3ParDefaut]
    preferenceParDefaut='Preference'

# Corps du programme
Debut
    Ecrire("Veuillez entrer une langue:")
    Lire(parametreUtilisateur.langue)
    Ecrire("Veuillez entrer le premier choix de couleur:")
    Lire(parametreUtilisateur.couleur.choix1)
    Ecrire("Veuillez entrer le deuxieme choix de couleur:")
    Lire(parametreUtilisateur.couleur.choix2)
    Ecrire("Veuillez entrer le troisieme choix de couleur:")
    Lire(parametreUtilisateur.couleur.choix3)
    Ecrire(parametreUtilisateur.couleur)
Fin

# Procédure pour configurer la langue
Procedure configLangue()
    si parametreUtilisateur.langue != langueParDefaut faire
        parametreUtilisateur.langue = langueSaisie
    sinon si parametreUtilisateur.langue = langueParDefaut faire
        parametreUtilisateur.langue = 'Anglais'
    sinon
        ecrire("La langue saisie est invalide")

# Procédure pour configurer la couleur
Procedure configCouleur()
    si parametreUtilisateur.couleur != couleurParDefaut faire
        parametreUtilisateur.couleur = couleurSaisie
    sinon si parametreUtilisateur.couleur = couleurParDefaut faire
        parametreUtilisateur.couleur = 'Rouge'
    sinon
        ecrire("La couleur saisie est invalide")

# Procédure pour configurer la préférence
Procedure configPreference()
    si parametreUtilisateur.preference != preferenceParDefaut faire
        parametreUtilisateur.preference = preferenceSaisie
    sinon si parametreUtilisateur.preference = preferenceParDefaut faire
        parametreUtilisateur.preference = 'Preference'
    sinon
        ecrire("La préférence saisie est invalide")
